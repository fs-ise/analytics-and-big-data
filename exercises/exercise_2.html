import numpy as np
import pandas as pd

rng = np.random.default_rng(42)

n = 200
df = pd.DataFrame({
    "age": rng.integers(18, 66, size=n),
    "department": rng.choice(["Sales", "Ops", "R&D"], size=n, p=[0.35, 0.40, 0.25]),
    "remote_days_per_week": rng.integers(0, 6, size=n),
})

# base salary with department effects + noise
dept_effect = df["department"].map({"Sales": 8000, "Ops": 5000, "R&D": 11000})
df["salary_eur"] = (35000 + dept_effect + 250 * df["age"] + rng.normal(0, 6000, size=n)).round(0)

# Inject a few missing values (common in real data)
missing_idx = rng.choice(df.index, size=8, replace=False)
df.loc[missing_idx[:4], "salary_eur"] = np.nan
df.loc[missing_idx[4:], "remote_days_per_week"] = np.nan

  • Group means ignore missing values unless you explicitly change that behavior.
